1boolean isMatch(String s, String p) {
2 int m=s.length(), n=p.length();
3 boolean[][] dp = new boolean[m+1][n+1];
4 dp[0][0] = true; // empty matches empty
5 for (int j=1; j<=n; j++) // x* can match empty
6 if (p[j-1]=='*') dp[0][j] = dp[0][j-2];
7 for (int i=1; i<=m; i++)
8 for (int j=1; j<=n; j++) {
9 if (p[j-1]=='.' || p[j-1]==s[i-1])
10 dp[i][j] = dp[i-1][j-1];
11 else if (p[j-1]=='*') {
12 dp[i][j] = dp[i][j-2]; // zero copies
13 if (p[j-2]=='.' || p[j-2]==s[i-1])
14 dp[i][j] |= dp[i-1][j]; // one more copy
15 }
16 }
17 return dp[m][n];
18}

Rules
'.' matches any single char.
x* matches zero or more of the char x before it.
* uses two paths: skip x* (dp[i][j-2]) or consume one x (dp[i-1][j]).
